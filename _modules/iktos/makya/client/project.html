def batch(iterable, n: int = 1):
    length = len(iterable)
    for ndx in range(0, length, n):
        yield iterable[ndx : min(ndx + n, length)]

    """
    Class for Makya Projects. Projects are created via an asynchronous method of class :py:class:`~iktos.makya.client.client.MakyaClient`.

    Args:
        client: Makya client. Not user specified
        graphql_id: GraphQL ID. Set by default. Not user specified
        slug: GraphQL slug. Not user specified
        name: Name of the project
        short_lived: If `False`, the project cannot be deleted from pyMakya
        makya_client: Makya client. Not user specified

    Examples:
        >>> from iktos.makya import MakyaClient
        >>> from iktos.makya.client.ipython_console import signon
        >>> client = MakyaClient(
        ...     base_url=makya_url,
        ...     email=makya_user_email,
        ...     password=makya_user_password,
        ... )
        >>> await signon(client)
        >>> new_project = await client.create_project("my-new-project", short_lived=True)
        >>> await new_project.delete_permanently()
    """

        self,
        client,
        graphql_id: str,
        slug: str,
        name: str,
        short_lived: bool = False,
        makya_client=None,
    ):
        self.client = client
        self.graphql_id = graphql_id
        self.slug = slug
        self.name = name
        self.short_lived = short_lived
        self.makya_client = makya_client

[docs]    async def delete_permanently(self) -> bool:
        """
        Deletes the project and all of its objects (datasets, predictors, scorers, generators, molecules).
        The data is deleted permanently with no way to restore it.
        Only `short_lived` projects can be deleted from pyMakya.

        Returns:
            bool: True if the project has been deleted, False otherwise
        """
        if self.short_lived:
            return await delete_project(self.client, self.graphql_id)

        logger.error("Only short lived projects can be deleted")
        return False

[docs]    async def create_dataset(  # noqa
        self,
        dataset_name: str,
        dataframe,
        smiles_col_name: str = 'Smiles',
        custom_id_col_name=None,
    ) -> Tuple[Optional[MakyaDataset], Any]:
        """
        Creates a new :py:class:`~iktos.makya.client.dataset.MakyaDataset` in the project.

        Args:
            dataset_name: Name of the dataset
            dataframe: Dataframe that is to be uploaded as a dataset
            smiles_col_name: Name of the SMILES column in the dataframe
            custom_id_col_name: Name of a custom compounds ID column

        Returns:
            :py:class:`~iktos.makya.client.dataset.MakyaDataset`, or (None, "ErrorMessage") if the creation failed.

        Notes:

            - The dataframe must contains a column of SMILES.
            - The dataframe must not contain:
                - chiral SMILES,
                - salts,
                - charged molecules,
                - columns that are not SMILES or custom ID and contain strings (ex: ">3", "N.A."...)
        """

        smiles_col_name = self._try_get_column_name_in_dataframe(
            dataframe, smiles_col_name
        )
        if smiles_col_name is None:
            return None, "The dataframe doesn't contain the requested SMILES column"

        if custom_id_col_name is not None:
            custom_id_col_name = self._try_get_column_name_in_dataframe(
                dataframe, custom_id_col_name
            )
            if custom_id_col_name is None:
                return (
                    None,
                    "The dataframe doesn't contain the requested custom id column",
                )

        dataset_with_same_name = await get_dataset_by_name(
            client=self.client, project_slug=self.slug, dataset_name=dataset_name
        )
        if dataset_with_same_name is not None:
            return (
                None,
                "A dataset with the same already exists, please change the name",
            )

        target_names = list()

        for target_name in dataframe.columns:
            if (
                target_name != smiles_col_name and target_name != custom_id_col_name
            ):  # noqa
                target_names.append(target_name)

        molecule_edges = []
        smiles_list = []

        for index, row in dataframe.iterrows():
            smiles = row[smiles_col_name]
            if smiles not in smiles_list:
                smiles_list.append(smiles)

            edge_dict: Dict[str, Any] = {'molecule': {'smiles': smiles}}
            if custom_id_col_name is not None:
                customId: str = str(row[custom_id_col_name])
                edge_dict['molecule']['customId'] = customId

            for target_name in target_names:
                target_dict = {'name': target_name}
                if type(row[target_name]) is str:
                    try:
                        value = float(row[target_name])
                        target_dict['value'] = value
                    except Exception:
                        target_dict['value'] = row[target_name]
                elif not np.isnan(row[target_name]):
                    target_dict['value'] = row[target_name]

                edge_dict.setdefault('targetsValue', []).append(target_dict)
            molecule_edges.append(edge_dict)

        result = await check_smiles(self.client, smiles_list)
        if result is None:
            return None, "Failed to check SMILES list"

        if not result["isValid"]:
            return None, result

        if result["chirals"]:
            return None, result

        if result["invalids"]:
            return None, result

        dataset, err = await create_dataset(
            client=self.client,
            project_id=self.graphql_id,
            dataset_name=dataset_name,
            target_names=target_names,
        )

        if err is not None or dataset is None:
            return None, err

        dataset_id = dataset['node']['id']
        dataset_slug = dataset['node']['slug']
        dataset_name = dataset['node']['name']

        for x in batch(molecule_edges, 100):
            result = await add_molecules_to_dataset(
                client=self.client, dataset_id=dataset_id, molecule_edges=x
            )

            assert (
                result is not None
                and result['addMoleculesValuesToDataset']['errors'] is None
            )

        await mark_dataset_uploaded(client=self.client, dataset_id=dataset_id)

        return (
            MakyaDataset(self.client, self, dataset_id, dataset_slug, dataset_name),
            None,
        )


    def _try_get_column_name_in_dataframe(self, dataframe, column_name):
        if column_name is not None:
            index = self._get_column_index(dataframe, column_name)
            if index == -1:
                logger.error(
                    f"The dataframe doesn't contain any column matching {column_name}"
                )
                return None

            return dataframe.columns[index]

        return None

    def _get_column_index(self, dataframe, column_name: str):
        results: List[bool] = dataframe.columns.str.match(
            f"^{column_name}$", case=False
        ).tolist()
        return results.index(True) if True in results else -1


[docs]    async def create_predictor_set(self, predictor_set_name, targets: List[dict] = []):
        """
        Creates a new :py:class:`~iktos.makya.client.predictor_set.MakyaPredictorSet` AutoML model in the project.

        Args:
            predictor_set_name: Name of the predictor_set
            targets: List of the targets (:py:class:`~iktos.makya.client.dataset.MakyaTarget`) and the associated value ranges for classification
                A target dictionary only supports arguments shown in the example below.

        Returns:
            :py:class:`~iktos.makya.client.predictor_set.MakyaPredictorSet`.

        Example:
            >>> dataset_target = await dataset.get_targets()
            >>> predictorset_targets = [
            ...     dict(
            ...         target=dataset_target['name_target_1'],
            ...         min=0.2,
            ...         max=dataset_target['name_target_1'].statistics['max'],
            ...     ),
            ...     dict(
            ...         target=dataset_target['name_target_2'],
            ...         min=dataset_target['name_target_1'].statistics['min'],
            ...         max=50,
            ...     ),
            ... ]
            >>> my_predictor_set = await project.create_predictor_set(
            ...     'my-predictor-set',
            ...     predictorset_targets
            ... )
        """

        predictor_set_dict = await create_predictor_set(
            client=self.client,
            project_id=self.graphql_id,
            predictor_set_name=predictor_set_name,
        )

        if predictor_set_dict is None:
            raise Exception(
                f'Predictor Set with name "{predictor_set_name}" already exists in project "{self.name}"'
            )  # noqa

        predictor_set = MakyaPredictorSet(
            client=self.client,
            project=self,
            graphql_id=predictor_set_dict['node']['id'],
            slug=predictor_set_dict['node']['slug'],
            name=predictor_set_dict['node']['name'],
        )

        if not targets:
            return predictor_set

        targets_dto = list()
        for target in targets:
            blueprint: Dict[str, Any] = {'limit': {}}

            if target['min'] is not None:
                blueprint['limit']['min'] = float(target['min'])

            if target['max'] is not None:
                blueprint['limit']['max'] = float(target['max'])

            if (
                'min' not in blueprint['limit'] and 'max' not in blueprint['limit']
            ):  # noqa
                raise Exception(f'A min or max is required in target blueprint')  # noqa

            targets_dto.append(
                {'blueprint': blueprint, 'id': target['target'].graphql_id}
            )

        result = await update_predictor_set(
            client=self.client,
            predictor_set_id=predictor_set.graphql_id,
            targets=targets_dto,
        )

        if result['updatePredictorSet']['errors'] is not None:
            raise Exception('Failed to create predictor set')

        return predictor_set

[docs]    async def create_generator(
        self, generator_name: str, configuration: GeneratorConfiguration
    ):
        """
        Creates a new :py:class:`~iktos.makya.client.generator.MakyaGenerator` in the project.

        Args:
            generator_name: Name of the generator
            configuration: Configuration to set up the generator
                The configuration depends on the type of generator. See the `iktos.makya.client.generator_configuration` documentation page.

        Returns:
            :py:class:`~iktos.makya.client.generator.MakyaGenerator`.

        """

        if configuration.generator_type is None:
            raise ValueError("Generator type not set")

        configuration_dict = configuration.to_dict()
        if configuration_dict is None:
            raise ValueError("Invalid generator configuration")

        engine = await self.__get_engine_by_id(configuration.generator_type)

        if engine is None:
            raise Exception('Generator engine unknown or unavailable')

        if engine.has_additional_parameters:
            await self.__validate_additional_parameters(
                configuration.additional_parameters, engine
            )

        generator_dict = await create_generator(
            client=self.client,
            project_id=self.graphql_id,
            generator_name=generator_name,
            generator_type=engine.id,
        )

        if generator_dict is None:
            raise Exception(
                f'Generator with name "{generator_name}" already exists in project "{self.name}"'
            )  # noqa

        generator = MakyaGenerator(
            client=self.client,
            project=self,
            graphql_id=generator_dict['node']['id'],
            slug=generator_dict['node']['slug'],
            name=generator_dict['node']['name'],
            engine=engine,
            configuration=configuration,
        )

        await generator.update_configuration(configuration)

        return generator

    async def __map_post_processor_scores(self, scores: list, name: str):
        scores_list: List[PostProcessorScore] = []
        if not scores:
            return scores_list

        for score in scores:
            if score["type"] == "DESCRIPTOR":
                scores_list.append(PostProcessorDescriptorScore(descriptor=score["id"]))
            elif score["type"] == "PREDICTOR":
                scores_list.append(
                    PostProcessorAutoMLScore(
                        predictor=await self.get_predictor(
                            to_global_id("Predictor", score["id"])
                        )
                    )
                )
            elif score["type"] == "EXTERNAL_SCORER":
                scores_list.append(
                    PostProcessorApiScore(
                        scorer=await self.get_external_scorer(
                            to_global_id("ExternalScorer", score["id"])
                        )
                    )
                )

        return scores_list
