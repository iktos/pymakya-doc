    def __init__(self, client, predictor_set, graphql_id, target, generator=None):
        self.client = client
        self.predictor_set = predictor_set
        self.graphql_id = graphql_id
        self.target = target
        self.generator = generator
        self.model = {}

    async def set_model(self, model):
        (type, predictor_id) = from_global_id(self.graphql_id)

        new_model = model.to_dict()

        for index in new_model["targets"]:
            new_model["targets"][index]["target_name"] = predictor_id
        self.model = json.dumps(new_model)
        await create_predictor_model(
            client=self.client,
            predictor_set_id=self.predictor_set.graphql_id,
            predictor_id=self.graphql_id,
            model=self.model,
        )

    async def get_model(self):
        model_json, target_name = await get_predictor_model(
            client=self.client,
            predictor_id=self.graphql_id,
        )

        if model_json is None:
            self.model = None
            return None

        model = json.loads(model_json)

        targets = model["targets"]
        target = list(targets.values())[0]
        key = list(targets.keys())[0]

        targets[target_name] = target
        del targets[key]

        target["target_name"] = target_name

        self.model = model
        return model

[docs]    async def get_predictors(self) -> dict:
        """
        Returns the Predictors (objectives) of the predictor set.

        Returns:
            dict: dictionary of the predictors
        """
        predictors = await get_predictors(
            client=self.client,
            project_slug=self.project.slug,
            predictor_set_slug=self.slug,
        )
        predictors_dict = dict()

        for predictor in predictors:
            name = predictor['node']['target']['name']
            target = dict(
                name=name, statistics=predictor['node']['target']['statistics']
            )

            predictors_dict[name] = MakyaPredictor(
                client=self.client,
                predictor_set=self,
                graphql_id=predictor['node']['id'],
                target=target,
            )

        return predictors_dict

[docs]    async def show_all_metrics(self):
        """
        Displays the metrics allowing the user to evaluate the performances of the predictor set.
        The metrics are displayed as an `ipywidgets` with one tab per objective of the model.
        In each tab, the following performance indicators are shown for both the train and test sets:

        - Precision: a measure of how many of the positive predictions made are correct (truly positive cases).
        - Recall: a measure of how many of the positive cases were correctly predicted by the classifier.
        - AUC: Area Under the Curve of the ROC (Receiver Operating Characteristic) curve, a measure of the ability of classifier to distinguish between classes.
        - F1-score: a score that combines precision and recall in a balanced way.
        - ROC curve: the basis for the calculation of the AUC. A plot of the False Positive Rate vs True Positive Rates at various thresholds.
        - Confusion matrix: a table that reports the number of true positives, true negatives, false positives and false negatives.
        """
        predictors = await self.get_predictors()

        out = []

        for _ in predictors:
            out.append(widgets.Output())

        tab = widgets.Tab(children=out)

        for i, target_name in enumerate(predictors):
            tab.set_title(i, target_name)

        display(tab)

        for i, target_name in enumerate(predictors):
            with out[i]:
                await predictors[target_name].show_metrics()

[docs]    async def show_progress(self):
        """
        Shows the progress of the model training as a progress bar.
        """
        result = await get_predictor_set(
            client=self.client, predictor_set_id=self.graphql_id
        )

        if result['status']['code'] == 'RUNNING':
            progress = result['progress']

            with tqdm_notebook(total=100) as pbar:
                pbar.update(int(progress))
                pbar.refresh()

                while progress < 100.0:
                    result = await get_predictor_set(
                        client=self.client, predictor_set_id=self.graphql_id
                    )

                    if result['status']['code'] != 'RUNNING':
                        return result['status']['code']

                    if int(result['progress']) - int(progress) > 0:
                        pbar.update(int(result['progress']) - int(progress))
                        pbar.refresh()
                        progress = result['progress']

                    try:
                        await asyncio.shield(asyncio.sleep(0.2))
                    except Exception:
                        raise Exception('Call the show_progress function again')

        else:
            print(result['status']['code'])



[docs]    async def predict(self, smiles_list: List[str]):
        """
        Applies the prediction model on a list of smiles.

        Args:
            smiles_list: List of achiral SMILES for prediction.

        Returns:
            a dataframe that contains the predicted scores for each objective and each SMILES.

        Examples:
            >>> smiles_list = [
            ...     'C1=CC(CCCc2ccccc2)C=C1',
            ...     'CCC',
            ... ]
            >>> predictions = await my_predictor_set.predict(smiles_list)
        """
        result = await get_predictor_set(
            client=self.client, predictor_set_id=self.graphql_id
        )

        if result is None:
            raise Exception("Can't retrieve the predictor set from the backend")

        if result['status']['code'] != 'DONE':
            raise Exception(
                "Can't predict now, the predictor set doesn't have a model yet"
            )

        batch_size = 100
        batches = math.ceil(len(smiles_list) / batch_size)

        results = []
        for batch_index in range(batches):
            batch = smiles_list[
                batch_index * batch_size : (batch_index + 1) * batch_size
            ]

            response = await predict(
                self.client, predictor_set_id=self.graphql_id, smiles_list=batch
            )

            results.extend([edge["node"] for edge in response])

        return self._predictions_to_dataframe(results)



[docs]    async def get_all_predicted_molecules(
        self,
        max_count: Optional[int] = None,
    ) -> MakyaMoleculesDataFrame:
        """
        Gets all the predictions made by a predictor set.

        Args:
            max_count: Maximum number of predictions to return.

        Returns:
            a dataframe that contains the predicted scores for each objective and each previously tested SMILES.

        Examples:
            >>> past_predictions = await my_predictor_set.get_all_predicted_molecules()
        """
        molecules_array: List[OrderedDict] = []
        after = None

        while max_count is None or len(molecules_array) < max_count:
            try:
                molecules = await get_predictor_set_all_predicted_molecules(
                    client=self.client,
                    project_slug=self.project.slug,
                    predictor_set_slug=self.slug,
                    after=after,
                    max_count=max_count - len(molecules_array)
                    if max_count is not None
                    else None,
                )

            except json.JSONDecodeError:
                continue
            except Exception as error:
                raise error

            if molecules is None or len(molecules) == 0:
                break

            molecules_array.extend([edge["node"] for edge in molecules])
            after = molecules[-1]['cursor']

        return self._predictions_to_dataframe(molecules_array)


    def _predictions_to_dataframe(self, predictions: List[OrderedDict]):
        if predictions is None or len(predictions) == 0:
            return MakyaMoleculesDataFrame([])

        datasource: dict = {"smiles": []}
        for prediction in predictions:
            datasource["smiles"].append(prediction["molecule"]["smiles"])
            for target_value in prediction["targetsValue"]:
                if target_value["target"]["name"] not in datasource:
                    datasource[target_value["target"]["name"]] = []

                datasource[target_value["target"]["name"]].append(target_value["value"])

        result = MakyaMoleculesDataFrame(datasource)
        return result
