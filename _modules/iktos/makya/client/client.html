def _make_endpoint_url(base_url: str, localhost_port: int, relative_url: str):
    if base_url.endswith('localhost'):
        return f'{base_url}:{localhost_port}{relative_url}'
    else:
        return f'{base_url}{relative_url}'


def _make_api_url(base_url: str, relative_url: str):
    return _make_endpoint_url(base_url, 8082, relative_url)


def _make_browser_url(base_url: str, relative_url: str):
    return _make_endpoint_url(base_url, 8080, relative_url)



[docs]class MakyaClient:
    """
    Class for Makya Clients.

    Args:
        endpoint: GraphQL endpoint. Deprecated, use base_url instead
        email: Mail address of the user's Makya account
        password: Password to the user's Makya account
        base_url: URL to the Makya web client
        aiohttp_session_extra: Options to configure the aiohttp module

    Examples:
        >>> from iktos.makya import MakyaClient
        >>> from iktos.makya.client.ipython_console import signon
        >>> client = MakyaClient(
        ...     base_url=makya_url,
        ...     email=makya_user_email,
        ...     password=makya_user_password,
        ... )
        >>> await signon(client)
    """

    def __init__(
        self,
        endpoint: Optional[str] = None,
        email: Optional[str] = None,
        password: Optional[str] = None,
        base_url=None,
        aiohttp_session_extra: dict = {},
    ):
        self.base_url = base_url
        self.endpoint = (
            endpoint
            if endpoint is not None
            else _make_api_url(self.base_url, '/api/graphql')
        )
        self.email = email
        self.password = password
        self.client = GraphQLClient(
            self.endpoint, '', aiohttp_session_extra=aiohttp_session_extra
        )

    async def connect(
        self,
        token: Optional[str] = None,
    ) -> Tuple[SignedOn, NeedsToken]:
        if self.email is None or self.password is None:
            raise Exception('Username or password missing')

        (need_token, cookie) = await login(
            self.client,
            self.email,
            self.password,
            int(token) if token is not None else None,
        )

        if not need_token and cookie is None:
            raise Exception('Failed to connect to server')

        if need_token:
            return (False, True)

        if cookie is None:
            raise Exception('Failed to connect to server')

        self.client.headers['cookie'] = cookie

        return (True, False)


[docs]    async def get_projects(self) -> dict:
        """
        Gets the projects associated with the Makya client.

        Returns:
            dict: A dictionary of the projects. Format: {`"name"`: :py:class:`~iktos.makya.client.project.MakyaProject`}
        """

        projects = await get_projects(self.client)
        projects_dict = {}

        for project in projects:
            node = project['node']

            projects_dict[node['name']] = self.__get_project_from_graphql_node(node)

        return projects_dict



[docs]    async def create_project(
        self, project_name: str, short_lived: bool = False
    ) -> MakyaProject:
        """
        Creates a new project in the Makya client.

        Args:
            project_name: Name of the new project
            short_lived: If `False`, the project cannot be deleted from pyMakya

        Examples:
            >>> new_project = await client.create_project("my-new-project", short_lived=True)
        """

        project = await create_project(
            client=self.client,
            project_name=project_name,
            short_lived=short_lived,
        )

        if project is None:
            raise Exception(
                f'Project with name "{project_name}" already exists'
            )  # noqa

        return self.__get_project_from_graphql_node(project['node'])



[docs]    async def get_possible_exit_vectors(self, smiles: str) -> Optional[List[float]]:
        """
        Computes the possible exit vectors to a given molecule.

        Args:
            smiles: SMILES of the molecule

        Examples:
            >>> from iktos.makya.client.ipython_console import draw_molecule_with_exit_vectors
            >>> from rdkit import Chem
            >>> ref_smiles = 'C1=CN(N=C1)N'
            >>> vectors = await client.get_possible_exit_vectors(ref_smiles)
            [[5],
            [0],
            [1],
            [4],
            [0, 5],
            [1, 5],
            [4, 5],
            [1, 2, 5],
            [2],
            [3],
            [0, 1],
            [0, 4],
            [1, 4],
            [2, 3],
            [0, 1, 4],
            [1, 2],
            [3, 4],
            [0, 1, 2, 3, 4],
            [0, 1, 5],
            [0, 4, 5],
            [1, 4, 5]]

            >>> # To display the possible exit vectors on the molecule:
            >>> draw_molecule_with_exit_vectors(Chem.MolFromSmiles(ref_smiles), vectors)
        """
        vectors = await get_possible_exit_vectors(client=self.client, smiles=smiles)

        if vectors is None:
            raise Exception("Couldn't find exit vectors")  # noqa

        return vectors

[docs]    def open_in_makya_browser(self, obj: Any = None):
        """
        Opens objects in the browser. The user needs to be signed into Makya in their browser.

        Args:
            obj: Object (Makya project, dataset, predictor set, generator, scorer...) to open in the Makya web client

        Examples:
            >>> client.open_in_makya_browser()

            >>> project = projects['my-project']
            >>> client.open_in_makya_browser(project)

            >>> datasets = await project.get_datasets()
            >>> dataset = datasets['my-dataset']
            >>> client.open_in_makya_browser(dataset)
        """

        if self.base_url is None:
            raise (Exception("Client's base url is not set"))

        if obj is None:
            obj = self

        if not hasattr(obj, 'get_permalink_url'):
            raise (Exception(f"This object of type {type(obj)} doesn't provide a url"))

        url = _make_browser_url(self.base_url, obj.get_permalink_url())
        webbrowser.open_new_tab(url)


    def __get_project_from_graphql_node(self, node):
        return MakyaProject(
            client=self.client,
            graphql_id=node['id'],
            slug=node['slug'],
            name=node['name'],
            short_lived=bool(node['shortLived']) is True
            if 'shortLived' in node
            else False,
            makya_client=self,
        )

    async def score_dataset_with_post_processor(
        self, post_processor: MakyaPostProcessor, dataset: MakyaDataset
    ):
        scored_dataset = await score_dataset_with_post_processor(
            self.client, post_processor.graphql_id, dataset.graphql_id
        )
        if scored_dataset:
            return MakyaDataset(
                self.client,
                dataset.project,
                scored_dataset["id"],
                scored_dataset["slug"],
                scored_dataset["name"],
            )
        else:
            return None
