[docs]class MakyaPostProcessor:
    """This class represents the post processor scores found in Makya
    Such a scorer can be used to aggregate multiple API, automl or descriptor targets and build a scoring function.
    The scoring function can then be used with generators to score generated molecules
    without impacting the reward function of the generation itself (hence the post processing part).
    They can also be used adhoc to compute scores for a dataset of molecules.

    Be aware that the scoring itself occurs in the background. So when you submit a dataset to score or run a generation
    the scores may take some time to be registered in you data, so you might need to refresh the resulting dataset a few times

    Args:
        client: Makya client. Not user specified
        project: Makya project. Not user specified
        graphql_id: GraphQL ID. Set by default. Not user specified
        name: Name of the Makya post-processor
        slug: GraphQL slug. Not user specified
        scores: List of scores that are aggregated in the scorer
        created_at: Creation date of the predictor set. Not user specified

    Example:
        >>> from iktos.makya.client.post_processor_configuration import (
        ...     PostProcessorApiScore,
        ...     PostProcessorAutoMLScore,
        ...     PostProcessorDescriptorScore
        ... )

        >>> # Print supported descriptors from RDKit
        >>> from iktos.makya.client.post_processor_configuration import valid_descriptor_names
        >>> valid_descriptor_names

        >>> # RDKit descriptor
        >>> score1 = PostProcessorDescriptorScore(descriptor="NumRings")
        >>> # API Scorer
        >>> external_scorers = await project.get_scorers()
        >>> score2 = PostProcessorApiScore(scorer=external_scorers['my API scorer'])
        >>> # AutoML predictor
        >>> predictor_sets = await project.get_predictor_sets()
        >>> predictor_set = predictor_sets['my predictor set']
        >>> predictors = await predictor_set.get_predictors()
        >>> score3 = PostProcessorAutoMLScore(predictor=predictors["Target1"])
        >>> # Creates the post-processing scorer
        >>> new_scorer = await project.create_post_processor(
        ...     name="my scorer",
        ...     scores=[
        ...         score1,
        ...         score2,
        ...         score3
        ...     ]
        ... )
    """

    def __init__(
        self,
        client: Any,
        project: Any,
        graphql_id: str,
        name: str,
        slug: str,
        scores: List[PostProcessorScore],
        created_at: Optional[str] = None,
    ):
        self.client = client
        self.project = project
        self.graphql_id: str = graphql_id
        self.name: str = name
        self.slug: str = slug
        self.scores: List[PostProcessorScore] = scores
        self.created_at = created_at


[docs]    def get_permalink_url(self):
        """
        Returns:
            str: The permalink to the post-processing scorer
        """
        return f'{self.project.get_permalink_url()}/post-processors/{self.slug}'


    def __repr__(self):
        return f"MakyaPostProcessor(name={self.name}, id={self.graphql_id})"
