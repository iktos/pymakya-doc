    """
    Class for Makya Generators. Generators are created via an asynchronous method of class :py:class:`~iktos.makya.client.project.MakyaProject`.

    Args:
        client: Makya client. Not user specified
        project: Makya project. Not user specified
        graphql_id: GraphQL ID. Set by default. Not user specified
        name: Name of the generator
        slug: GraphQL slug. Not user specified
        engine: Makya generator engine. Not user specified
        configuration: Configuration to set up the generator
                The configuration depends on the type of generator. See the `iktos.makya.client.generator_configuration` documentation page.
        created_at: Creation date of the generator. Set automatically

    Examples:
        >>> from iktos.makya.client.generator_configuration import NoveltyGeneratorConfiguration, MatchSubstructureConstraint
        >>> goals = [
        ...     MatchSubstructureConstraint(smarts="CC"),
        ... ]
        >>> configuration = NoveltyGeneratorConfiguration(
        ...     reference_molecule_smiles='Nc1ncnc2[nH]cnc12',
        ...     substructure_constraints=goals,
        ...     additional_parameters={}
        ... )
        >>> my_generator = await project.create_generator(generator_name="my-novelty-generator", configuration=configuration)
        >>> await my_generator.start()
    """

    def __init__(
        self,
        client,
        project,
        graphql_id: str,
        name: str,
        slug: str,
        engine: MakyaGeneratorEngine,
        configuration: GeneratorConfiguration,
        created_at: Optional[str] = None,
    ):
        self.client = client
        self.project = project
        self.graphql_id = graphql_id
        self.name = name
        self.slug = slug
        self.engine = engine
        self.configuration = configuration
        self.created_at = created_at

        """
        Returns the generator's datasets.

        Returns:
            dict: A dictionary of the datasets. Format: {`"name"`: :py:class:`~iktos.makya.client.dataset.MakyaDataset`}
        """

        datasets = await get_generator_datasets(
            client=self.client, project_slug=self.project.slug, generator_slug=self.slug
        )
        datasets_dict = {}

        for dataset in datasets:
            node = dataset['node']
            datasets_dict[node['name']] = MakyaDataset(
                client=self.client,
                graphql_id=node['id'],
                slug=node['slug'],
                name=node['name'],
                project=self.project,
            )

        return datasets_dict

[docs]    async def get_all_generated_molecules(  # noqa: C901
        self,
        score_filters: Optional[
            Dict[GeneratorScoreFilterName, List[Optional[float]]]
        ] = None,
        max_count: Optional[int] = None,
    ) -> MakyaMoleculesDataFrame:
        """
        Returns the molecules generated by the generator.

        Args:
            score_filters: Dictionary of score filters.
                Only the molecules whose scores are in the specified range will be returned.
                GeneratorScoreFilterName: 'goal_score', 'confidence', 'qed', 'quality', 'similarity'.
                If the user provides a range ([a,b]), the returned molecules' scores are within this range.
                If the user provides a single value ([a]), the returned molecules' scores are greater than or equal to this value.
            max_count: Maximum number of molecules to return.

        Returns:
            a dataframe that contains the generated SMILES and their scores.

        Examples:
            >>> # To return molecules with a goal score >= 8:
            >>> generated_molecules = await my_generator.get_all_generated_molecules(score_filters={'goal_score': [8])
        """

        molecules_array: List[OrderedDict] = []
        step = 0
        after = None

        while max_count is None or len(molecules_array) < max_count:
            try:
                molecules = await get_generator_all_generated_molecules(
                    client=self.client,
                    project_slug=self.project.slug,
                    generator_slug=self.slug,
                    score_filters=score_filters,
                    after=after,
                    max_count=max_count - len(molecules_array)
                    if max_count is not None
                    else None,
                )

            except JSONDecodeError:
                continue
            except Exception as error:
                raise error

            if molecules is None or len(molecules) == 0:
                break

            for molecule in molecules:
                molecule_dict = OrderedDict()
                molecule_dict['Smiles'] = molecule['node']['molecule'][
                    'smiles'
                ]  # noqa: E501
                if (
                    'reactionSmiles' in molecule['node']['molecule']
                    and molecule['node']['molecule']['reactionSmiles'] is not None
                ):
                    smiles = molecule['node']['molecule'][
                        'reactionSmiles'
                    ]  # noqa: E501
                    if '>>' in smiles:
                        try:
                            smiles = smiles.split('.', maxsplit=1)[1].split('>>')[0]
                        except Exception:
                            pass

                    molecule_dict['ReactionCompoundSmiles'] = smiles

                molecule_dict['QED'] = molecule['node']['score']['qed']
                molecule_dict['Confidence'] = molecule['node']['score'][
                    'confidence'
                ]  # noqa: E501
                molecule_dict['Quality'] = molecule['node']['score']['quality']
                molecule_dict['Similarity'] = molecule['node']['score'][
                    'similarity'
                ]  # noqa: E501
                molecule_dict['Goal Score'] = molecule['node']['score'][
                    'blueprintScore'
                ]  # noqa: E501
                after = molecule['cursor']

                for targetValue in molecule['node']['targetsValue']:
                    if targetValue['target']['customName'] is not None:
                        name = targetValue['target']['customName']

                    else:
                        name = targetValue['target']['name']

                    molecule_dict[name] = targetValue['value']

                for externalScorerValue in (
                    molecule['node']['externalScorersValues'] or []
                ):  # noqa: E501
                    for value in externalScorerValue['values']:
                        molecule_dict[value['targetName']] = value[
                            'value'
                        ]  # noqa: E501

                molecules_array.append(molecule_dict)
            step += 1
        molecules_df = DataFrame(molecules_array).dropna(axis=1, how='all')

        if len(molecules_array) > 0:
            PandasTools.AddMoleculeColumnToFrame(
                molecules_df, 'Smiles', 'Molecule', includeFingerprints=False
            )

            if 'ReactionCompoundSmiles' in molecules_df.columns:
                PandasTools.AddMoleculeColumnToFrame(
                    molecules_df,
                    'ReactionCompoundSmiles',
                    'ReactionCompound',
                    includeFingerprints=False,
                )
                cols = molecules_df.columns.tolist()
                cols = cols[-2:] + cols[:-2]
            else:
                cols = molecules_df.columns.tolist()
                cols = cols[-1:] + cols[:-1]

            molecules_df = molecules_df[cols]

        return MakyaMoleculesDataFrame(molecules_df.drop_duplicates(subset='Smiles'))

[docs]    async def show_progress(self):
        """
        Shows the progress of the generation as a progress bar.
        """

        result = await get_generator(client=self.client, generator_id=self.graphql_id)

        if result['status'] == 'RUNNING':
            progress = result['worker']['progress']

            with tqdm_notebook(total=100) as pbar:
                pbar.update(int(progress))
                pbar.refresh()

                while progress < 100.0:
                    result = await get_generator(
                        client=self.client, generator_id=self.graphql_id
                    )

                    if result['status'] != 'RUNNING':
                        return result['status']

                    if int(result['worker']['progress']) - int(progress) > 0:
                        pbar.update(int(result['worker']['progress']) - int(progress))
                        pbar.refresh()
                        progress = result['worker']['progress']

                    try:
                        await asyncio.shield(asyncio.sleep(0.2))
                    except Exception:
                        raise Exception('Call the show_progress function again')

        else:
            print(result['status'])



[docs]    async def get_full_configuration(self) -> dict:
        """
        Returns the full configuration of the generator as dictionary.
        """

        config = await get_full_configuration(
            client=self.client, generator_id=self.graphql_id
        )
        if config is None:
            config = {}

        config['datasets'] = [
            MakyaDataset(
                client=self.client,
                graphql_id=dataset['node']['id'],
                name=dataset['node']['name'],
                slug=dataset['node']['slug'],
                project=self.project,
            )
            for dataset in config['datasets']['edges']
        ]

        config['externalScorers'] = [
            MakyaScorer(
                client=self.client,
                project=self.project,
                graphql_id=scorer['id'],
                name=scorer['name'],
                slug=scorer['slug'],
                api_url=scorer['apiUrl'],
                target_configs_list=scorer['targetConfigsList'],
            )
            for scorer in config['externalScorers']
        ]

        config['additionalParameters'] = config['additionalParameters'] or dict()

        config['predictors'] = await self.get_predictors()

        return config



[docs]    async def get_predictors(self) -> Dict[str, MakyaPredictor]:
        """
        Returns the generator's predictors. Predictors are individual objectives of predictor sets.

        Returns:
            a dictionary of the predictors.
        """

        predictors = await get_predictors(
            client=self.client, project_slug=self.project.slug, generator_slug=self.slug
        )
        predictors_dict = dict()

        for predictor in predictors:
            name = predictor['node']['target']['name']
            target = dict(
                name=name, statistics=predictor['node']['target']['statistics']
            )

            predictors_dict[name] = MakyaPredictor(
                client=self.client,
                predictor_set=None,
                graphql_id=predictor['node']['id'],
                target=target,
                generator=self,
            )

        return predictors_dict



[docs]    async def push_molecules(self, molecules: List[Generation]):
        """
        Pushes molecules into a generator.
        This method can be useful to visualise external molecules in Makya and compare them with generated molecules.

        Args:
            molecules: List of Generation objects that associates molecules with scores.

        Examples:
            >>> from iktos.makya.client.generation.generation import MakyaMolecule, Generation, GenerationScore, CustomGenerationScore
            >>> from rdkit.Chem.QED import qed
            >>> from rdkit import Chem
            >>> smiles = 'CCCCC'
            >>> mol = Chem.MolFromSmiles(smiles)
            >>> molecule = MakyaMolecule(
            ...     smiles=smiles,
            ...     inchi=Chem.MolToInchi(mol, options="-Snon -FixedH"),
            ...     inchi_key=Chem.InchiToInchiKey(inchi),
            ... )
            >>> score = GenerationScore(
            ...     qed=qed(mol),
            ...     confidence=0.98,
            ...     custom_scores=[
            ...         CustomGenerationScore(score_name='IC50', value=32),
            ...         CustomGenerationScore(score_name='Volume', value=270),
            ...     ]
            ... )
            >>> generations = [Generation(
            ...     molecule=molecule,
            ...     score=score)
            ... ]
            >>> result = await my_generator.push_molecules(generations)
        """

        batch_size = 100
        batches = math.ceil(len(molecules) / batch_size)

        molecule_count = 0
        for batch_index in range(batches):
            batch = molecules[batch_index * batch_size : (batch_index + 1) * batch_size]

            molecule_count = await add_molecules_values_to_generator(
                self.client,
                generator_id=self.graphql_id,
                generation=[mol.to_dict() for mol in batch],
            )

        return molecule_count
