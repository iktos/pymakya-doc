[docs]class MakyaScorer:
    """
    Class for Makya API configurations. API scorers are created via an asynchronous method of class :py:class:`~iktos.makya.client.project.MakyaProject`.

    Args:
        client: Makya client. Not user specified
        project: Makya project. Not user specified
        graphql_id: GraphQL ID. Set by default. Not user specified
        name: API configuration name
        slug: GraphQL slug. Not user specified
        api_url: URL to set up the API. See Makya documentation for more details

    Examples:
        >>> config = [
        ...     dict(hardChecker=True,
        ...         index=23, targetName="IC50",
        ...         lowerThreshold=50, upperThreshold=None
        ...     ),
        ...     dict(hardChecker=False,
        ...         index=12, targetName="MolWT",
        ...         lowerThreshold=450, upperThreshold=None
        ...     ),
        ... ]
        >>> new_api_scorer = await my_project.create_scorer("my_scorer",
        ...     "http://localhost:4444",
        ...     config,
        ... )

    Notes:
        - The API specifications and set up is very technical.
            It is addressed in detail in a separate documentation dedicated to API.
            Kindly get in touch with your contact person at Iktos, or send us a message at support@makya.ai from your Makya registered email address.
    """

    def __init__(
        self,
        client,
        project,
        graphql_id: str,
        name: str,
        slug: str,
        api_url: str,
        target_configs_list: List[dict],
    ):
        self.client = client
        self.project = project
        self.graphql_id = graphql_id
        self.name = name
        self.slug = slug
        self.api_url = api_url
        self.target_configs_list = target_configs_list
