class NoveltyGeneratorConfiguration(GeneratorConfiguration):
    """
    Configuration class for a Novelty generator. Needed to create a generator.

    Args:
        reference_molecule_smiles: SMILES of the reference molecule from which patent hopping is performed.
        generator_type: Set by default. Not user specified
        additional_parameters: Additional parameters for generator configuration.
        datasets: Datasets to be used as chemical space.
            Can be a single molecule.
        predictors: Predictors (predictor sets's objectives) to guide generation.
        substructure_constraints: List of substructure constraints on the generated molecules.
            Set with the class :py:class:`~iktos.makya.client.generator_configuration.filters.SubstructConstraint`.
        external_scorers: List of API configurations to score the generated molecules.

    Examples:
        >>> from iktos.makya.client.generator_configuration import NoveltyGeneratorConfiguration, MatchSubstructureConstraint
        >>> goals = [
        ...     MatchSubstructureConstraint(smarts="CC"),
        ... ]
        >>> configuration = NoveltyGeneratorConfiguration(
        ...     reference_molecule_smiles='Nc1ncnc2[nH]cnc12',
        ...     substructure_constraints=goals,
        ... )
        >>> generator = await project.create_generator(generator_name="my-novelty", configuration=configuration)
    """

    reference_molecule_smiles: Optional[str] = None

    def __post_init__(self):
        super().__init__(
            generator_type="isostere",
            additional_parameters=self.additional_parameters,
            datasets=None,
            predictors=self.predictors,
            substructure_constraints=self.substructure_constraints,
            external_scorers=self.external_scorers,
        )

    def to_dict(self):
        # Chemical space isn't supported by the Novelty so just ignore it for now
        self.datasets = None
        result = super().to_dict()
        result['baseSmiles'] = self.reference_molecule_smiles
        return result
