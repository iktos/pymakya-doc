class CompoundFilter:
    """
    Class for filters on molecules or building blocks.

    Args:
        filter_name: Name of the filter to apply.
        min: Minimum value.
        max: Maximum value.

    Examples:
        >>> from iktos.makya.client.generator_configuration import CompoundFilter
        >>> commercial_compounds_filters = [
        ...     CompoundFilter(filter_name='MolWt', min=450, max=600),
        ...     CompoundFilter(filter_name='TPSA', min=0.33),
        ... ]
    """

    filter_name: CompoundFilterNames
    min: Optional[float] = None
    max: Optional[float] = None

    def to_dict(self):
        return {'filterName': self.filter_name, 'min': self.min, 'max': self.max}

class SubstructConstraint:
    """
    Class for substructure constraints.

    Args:
        smarts: SMARTS of the substructure(s) that are required or forbidden.
        type: Specify if the substructures are required (SUBSTRUCT_MATCH) or forbidden (SUBSTRUCT_UNMATCH).

    Examples:
        >>> from iktos.makya.client.generator_configuration import SubstructConstraint
        >>> commercial_compound_substruct_constraints = [
        ...     SubstructConstraint(
        ...         type='SUBSTRUCT_MATCH',
        ...         smarts='*~[#6]~[#6](=O)-[#8][H]'
        ...     )
        ... ]
    """

    type: Literal['SUBSTRUCT_MATCH', 'SUBSTRUCT_UNMATCH']
    smarts: str

    def to_dict(self):
        return {'type': self.type, 'smarts': self.smarts}

class ExitVectorsParams:
    """
    Class for exit vectors to specify on fragments (for the fragment growing and fragment linking generators).

    Args:
        smiles: SMILES of the fragment.
        exit_vector: List of exit vectors to be used, identified by a number.
            To compute possible exit vectors, use the :py:meth:`~iktos.makya.client.client.MakyaClient.get_possible_exit_vectors` method.
            To draw possible exit vectors, use the :py:meth:`~iktos.makya.client.ipython_console.draw_molecule_with_exit_vectors` function.
        commercial_compound_filters: Filters on fragment.
        commercial_compound_substruct_constraints: list of substructure constraints on the building blocks used for growing/linking.

    Examples:
        >>> reference_smiles_1 = 'Nc1ncnc2[nH]cnc12'

        >>> from iktos.makya.client.ipython_console import draw_molecule_with_exit_vectors
        >>> from rdkit import Chem
        >>> vectors_1 = vectors = await client.get_possible_exit_vectors(reference_smiles_1)
        >>> draw_molecule_with_exit_vectors(Chem.MolFromSmiles(reference_smiles_1), vectors_1)

        >>> from iktos.makya.client.generator_configuration import (
        ...     CompoundFilter,
        ...     ExitVectorsParams,
        ...     SubstructConstraint
        ... )
        >>> commercial_compounds_filters = [
        ...     CompoundFilter(filter_name='MolWt', min=450, max=600),
        ...     CompoundFilter(filter_name='TPSA', min=0.33),
        ... ]
        >>> commercial_compound_substruct_constraints = [
        ...     SubstructConstraint(
        ...         type='SUBSTRUCT_MATCH',
        ...         smarts='*~[#6]~[#6](=O)-[#8][H]'
        ...     )
        ... ]
        >>> exit_vector_1 = ExitVectorsParams(
        ...     smiles=reference_smiles_1,
        ...     exit_vector=[0],
        ...     commercial_compound_filters=commercial_compounds_filters,
        ...     commercial_compound_substruct_constraints=commercial_compound_substruct_constraints
        ... )
    """

    smiles: Optional[str]
    exit_vector: Optional[List[int]]
    commercial_compound_filters: Optional[List[CompoundFilter]] = None
    commercial_compound_substruct_constraints: Optional[
        List[SubstructConstraint]
    ] = None
