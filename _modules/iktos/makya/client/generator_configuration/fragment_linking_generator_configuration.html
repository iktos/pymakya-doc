class FragmentLinkingGeneratorConfiguration(GeneratorConfiguration):
    """
    Configuration class for a Fragment Linking generator. Needed to create a generator.

    Args:
        exit_vector_params: Fragment exit vectors.
            Specify 2 exit vectors for the 2 fragments to be linked.
            To compute possible exit vectors, use the :py:meth:`~iktos.makya.client.client.MakyaClient.get_possible_exit_vectors` method.
            To draw possible exit vectors, use the :py:meth:`~iktos.makya.client.ipython_console.draw_molecule_with_exit_vectors` function.
        final_product_filters: Filters on final generated molecules.
        generator_type: Set by default. Not user specified
        additional_parameters: Additional parameters for generator configuration.
        datasets: Datasets to be used as chemical space.
            Can be a single molecule.
        predictors: Predictors (predictor sets's objectives) to guide generation.
        substructure_constraints: List of substructure constraints on the generated molecules.
            Set with the class :py:class:`~iktos.makya.client.generator_configuration.filters.SubstructConstraint`.
        external_scorers: List of API configurations to score the generated molecules.

    Examples:
        >>> reference_smiles_1 = 'Nc1ncnc2[nH]cnc12'
        >>> reference_smiles_2 = 'Nc1ncnc2[nH]cnc12'

        >>> from iktos.makya.client.ipython_console import draw_molecule_with_exit_vectors
        >>> from rdkit import Chem
        >>> vectors_1 = vectors = await client.get_possible_exit_vectors(reference_smiles_1)
        >>> draw_molecule_with_exit_vectors(Chem.MolFromSmiles(reference_smiles_1), vectors_1)
        >>> vectors_2 = vectors = await client.get_possible_exit_vectors(reference_smiles_2)
        >>> draw_molecule_with_exit_vectors(Chem.MolFromSmiles(reference_smiles_2), vectors_2)

        >>> from iktos.makya.client.generator_configuration import (
        ...     FragmentLinkingGeneratorConfiguration,
        ...     CompoundFilter,
        ...     ExitVectorsParams,
        ...     SubstructConstraint
        ... )
        >>> commercial_compounds_filters = [
        ...     CompoundFilter(filter_name='MolWt', min=450, max=600),
        ...     CompoundFilter(filter_name='TPSA', min=0.33),
        ... ]
        >>> product_filters = [
        ...     CompoundFilter(filter_name='QED', min=0.7),
        ...     CompoundFilter(filter_name='NumAromaticRings', max=3),
        ... ]
        >>> commercial_compound_substruct_constraints = [
        ...     SubstructConstraint(
        ...         type='SUBSTRUCT_MATCH',
        ...         smarts='*~[#6]~[#6](=O)-[#8][H]'
        ...     )
        ... ]
        >>> exit_vector_1 = ExitVectorsParams(
        ...     smiles=reference_smiles_1,
        ...     exit_vector=[0],
        ...     commercial_compound_filters=commercial_compounds_filters,
        ...     commercial_compound_substruct_constraints=commercial_compound_substruct_constraints
        ... )
        >>> exit_vector_2 = ExitVectorsParams(
        ...     smiles=reference_smiles_2,
        ...     exit_vector=[6]
        ... )
        >>> configuration = FragmentLinkingGeneratorConfiguration(
        ...     exit_vector_params=[exit_vector_1, exit_vector_2],
        ...     final_product_filters=product_filters
        ... )
        >>> generator = project.create_generator(generator_name="my-fragment-linking", configuration=configuration)
    """

    exit_vector_params: Optional[List[ExitVectorsParams]] = None
    final_product_filters: Optional[List[CompoundFilter]] = None

    def __post_init__(self):
        super().__init__(
            generator_type="fragments-linking",
            additional_parameters=self.additional_parameters,
            datasets=self.datasets,
            predictors=self.predictors,
            substructure_constraints=self.substructure_constraints,
            external_scorers=self.external_scorers,
        )

    def to_dict(self):
        result = super().to_dict()
        result['forwardGenerationParameters'] = {
            'exitVectorsParams': [
                {
                    'smiles': param.smiles,
                    'exitVector': param.exit_vector,
                    'commercialCompoundsDefaultFilters': [
                        filter.to_dict() for filter in param.commercial_compound_filters
                    ]
                    if param.commercial_compound_filters is not None
                    else [],
                    'commercialCompoundsSubstructConstraints': [
                        constraint.to_dict()
                        for constraint in param.commercial_compound_substruct_constraints
                    ]
                    if param.commercial_compound_substruct_constraints is not None
                    else [],
                    'maxBatchSize': 100,
                }
                for param in self.exit_vector_params
            ],
            'productMolsFilters': [
                filter.to_dict() for filter in self.final_product_filters
            ]
            if self.final_product_filters is not None
            else [],
            # 'synpleChem': True,
        }

        return result
